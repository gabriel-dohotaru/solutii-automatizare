  <project_name>Soluții Automatizare - Software Automation Services Platform</project_name>
  <domain>solutiiautomatizare.ro</domain>

  <overview>
    Build a professional website and client portal for a software automation services company.
    The company specializes in:
    - E-commerce modules development (PrestaShop, WooCommerce, Magento, OpenCart)
    - Software automation (scripts, API integrations, process automation)
    - Bug fixing and maintenance services
    - Custom web development (websites, web applications, SaaS)

    The platform should showcase services, display pricing packages, allow quote requests,
    provide a client portal for project tracking, and feature a blog for SEO and expertise
    demonstration. Design should be modern, professional, and tech-oriented.
  </overview>

  <!-- ==================== DATABASE SCHEMA ==================== -->

  <database_schema>
    <tables>
      <users>
        - id INTEGER PRIMARY KEY
        - email TEXT UNIQUE NOT NULL
        - password_hash TEXT NOT NULL
        - first_name TEXT
        - last_name TEXT
        - company_name TEXT
        - phone TEXT
        - role TEXT DEFAULT 'client' (client/admin)
        - email_verified INTEGER DEFAULT 0
        - created_at DATETIME
        - updated_at DATETIME
      </users>

      <services>
        - id INTEGER PRIMARY KEY
        - name TEXT NOT NULL
        - slug TEXT UNIQUE
        - category TEXT (ecommerce/automation/bugfix/webdev)
        - short_description TEXT
        - full_description TEXT
        - icon TEXT
        - features JSON
        - is_featured INTEGER DEFAULT 0
        - order_index INTEGER
        - is_active INTEGER DEFAULT 1
      </services>

      <packages>
        - id INTEGER PRIMARY KEY
        - name TEXT NOT NULL (Starter/Professional/Enterprise)
        - slug TEXT UNIQUE
        - description TEXT
        - price_from DECIMAL(10,2)
        - currency TEXT DEFAULT 'EUR'
        - features JSON
        - is_popular INTEGER DEFAULT 0
        - order_index INTEGER
        - is_active INTEGER DEFAULT 1
      </packages>

      <projects>
        - id INTEGER PRIMARY KEY
        - client_id INTEGER REFERENCES users(id)
        - name TEXT NOT NULL
        - description TEXT
        - service_type TEXT
        - status TEXT (inquiry/approved/in_progress/review/completed/cancelled)
        - progress INTEGER DEFAULT 0
        - estimated_hours INTEGER
        - price DECIMAL(10,2)
        - currency TEXT DEFAULT 'EUR'
        - start_date DATE
        - deadline DATE
        - completed_date DATE
        - created_at DATETIME
        - updated_at DATETIME
      </projects>

      <project_milestones>
        - id INTEGER PRIMARY KEY
        - project_id INTEGER REFERENCES projects(id)
        - title TEXT
        - description TEXT
        - status TEXT (pending/in_progress/completed)
        - due_date DATE
        - completed_date DATE
        - order_index INTEGER
      </project_milestones>

      <project_updates>
        - id INTEGER PRIMARY KEY
        - project_id INTEGER REFERENCES projects(id)
        - user_id INTEGER REFERENCES users(id)
        - title TEXT
        - content TEXT
        - is_internal INTEGER DEFAULT 0
        - created_at DATETIME
      </project_updates>

      <project_files>
        - id INTEGER PRIMARY KEY
        - project_id INTEGER REFERENCES projects(id)
        - uploaded_by INTEGER REFERENCES users(id)
        - filename TEXT
        - original_name TEXT
        - file_type TEXT
        - file_size INTEGER
        - description TEXT
        - is_deliverable INTEGER DEFAULT 0
        - created_at DATETIME
      </project_files>

      <portfolio_items>
        - id INTEGER PRIMARY KEY
        - title TEXT NOT NULL
        - slug TEXT UNIQUE
        - category TEXT
        - client_name TEXT
        - short_description TEXT
        - challenge TEXT
        - solution TEXT
        - results TEXT
        - technologies JSON
        - images JSON
        - testimonial TEXT
        - is_featured INTEGER DEFAULT 0
        - is_published INTEGER DEFAULT 1
        - created_at DATETIME
      </portfolio_items>

      <blog_posts>
        - id INTEGER PRIMARY KEY
        - title TEXT NOT NULL
        - slug TEXT UNIQUE
        - excerpt TEXT
        - content TEXT
        - featured_image TEXT
        - category TEXT (tutorials/news/case-studies/tips)
        - author_id INTEGER REFERENCES users(id)
        - status TEXT (draft/published)
        - reading_time INTEGER
        - views INTEGER DEFAULT 0
        - published_at DATETIME
        - created_at DATETIME
        - updated_at DATETIME
      </blog_posts>

      <support_tickets>
        - id INTEGER PRIMARY KEY
        - user_id INTEGER REFERENCES users(id)
        - project_id INTEGER REFERENCES projects(id)
        - subject TEXT
        - status TEXT (open/in_progress/waiting/resolved/closed)
        - priority TEXT (low/medium/high/urgent)
        - created_at DATETIME
        - updated_at DATETIME
        - resolved_at DATETIME
      </support_tickets>

      <ticket_messages>
        - id INTEGER PRIMARY KEY
        - ticket_id INTEGER REFERENCES support_tickets(id)
        - user_id INTEGER REFERENCES users(id)
        - message TEXT
        - attachments JSON
        - is_internal INTEGER DEFAULT 0
        - created_at DATETIME
      </ticket_messages>

      <contact_submissions>
        - id INTEGER PRIMARY KEY
        - name TEXT
        - email TEXT
        - phone TEXT
        - company TEXT
        - service_type TEXT
        - budget_range TEXT
        - message TEXT
        - attachment_path TEXT
        - status TEXT (new/contacted/converted/closed)
        - notes TEXT
        - created_at DATETIME
        - updated_at DATETIME
      </contact_submissions>

      <quote_requests>
        - id INTEGER PRIMARY KEY
        - name TEXT
        - email TEXT
        - phone TEXT
        - company TEXT
        - project_type TEXT
        - platform TEXT (prestashop/woocommerce/magento/other)
        - description TEXT
        - timeline TEXT
        - budget_range TEXT
        - attachments JSON
        - status TEXT (new/reviewing/quoted/accepted/rejected)
        - quoted_price DECIMAL(10,2)
        - notes TEXT
        - created_at DATETIME
        - updated_at DATETIME
      </quote_requests>

      <invoices>
        - id INTEGER PRIMARY KEY
        - project_id INTEGER REFERENCES projects(id)
        - client_id INTEGER REFERENCES users(id)
        - invoice_number TEXT UNIQUE
        - amount DECIMAL(10,2)
        - currency TEXT DEFAULT 'EUR'
        - status TEXT (draft/sent/paid/overdue/cancelled)
        - due_date DATE
        - paid_date DATE
        - pdf_path TEXT
        - created_at DATETIME
      </invoices>

      <newsletter_subscribers>
        - id INTEGER PRIMARY KEY
        - email TEXT UNIQUE
        - is_active INTEGER DEFAULT 1
        - subscribed_at DATETIME
      </newsletter_subscribers>

      <sessions>
        - id INTEGER PRIMARY KEY
        - user_id INTEGER REFERENCES users(id)
        - token TEXT UNIQUE
        - expires_at DATETIME
        - created_at DATETIME
      </sessions>
    </tables>
  </database_schema>
